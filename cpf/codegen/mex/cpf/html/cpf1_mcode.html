1function [max_lambda, predicted_list, corrected_list, combined_list, success, et] = cpf(casedata, participation, sigmaForLambda, sigmaForVoltage, verbose)
  2%CPF  Run continuation power flow (CPF) solver.
  3%   [INPUT PARAMETERS]
  4%   loadvarloc: load variation location(in external bus numbering). Single
  5%               bus supported so far.
  6%   sigmaForLambda: stepsize for lambda
  7%   sigmaForVoltage: stepsize for voltage
  8%   [OUTPUT PARAMETERS]
  9%   max_lambda: the lambda in p.u. w.r.t. baseMVA at (or near) the nose
 10%               point of PV curve
 11%   NOTE: the first column in return parameters 'predicted_list,
 12%   corrected_list, combined_list' is bus number; the last row is lambda.

 46%% define named indices into bus, gen, branch matrices
 47
 48%escalate 'singular' to a matrix so we can use error handling to deal with
 49%it
 50
 51% [PQ, PV, REF, NONE, BUS_I, BUS_TYPE, PD, QD, GS, BS, BUS_AREA, VM, ...
 52%     VA, BASE_KV, ZONE, VMAX, VMIN, LAM_P, LAM_Q, MU_VMAX, MU_VMIN] = idx_bus();
 53% 
 54% [F_BUS, T_BUS, BR_R, BR_X, BR_B, RATE_A, RATE_B, ...
 55%     RATE_C, TAP, SHIFT, BR_STATUS, PF, QF, PT, QT, MU_SF, MU_ST] = idx_brch();
 56% [GEN_BUS, PG, QG, QMAX, QMIN, VG, MBASE, ...
 57%     GEN_STATUS, PMAX, PMIN, MU_PMAX, MU_PMIN, MU_QMAX, MU_QMIN] = idx_gen();
 58
 59BUS_I = 1;
 60PD = 3;
 61QD = 4;
 62VA = 9;
 63
 64GEN_BUS = 1;
 65VG = 6;
 66GEN_STATUS = 8;
 67
 68
 69
 70lastwarn('No Warning');
 71%% assign default parameters
 72if nargin < 3
 73    sigmaForLambda = 0.1;       % stepsize for lambda
 74    sigmaForVoltage = 0.025;    % stepsize for voltage
 75end
 76
 77if nargin < 5, verbose = 0; end
 78
 79
 80
 81%% options
 82max_iter = 1000;                 % depends on selection of stepsizes
 83
 84%% ...we use PV curve slopes as the criteria for switching modes
 85slopeThresh_Phase1 = 0.5;       % PV curve slope shreshold for voltage prediction-correction (with lambda increasing)
 86slopeThresh_Phase2 = 0.3;       % PV curve slope shreshold for lambda prediction-correction
 87
 88
 89%% load the case & convert to internal bus numbering
 90[baseMVA, bus, gen, branch] = loadcase(casedata);
 91[numBuses, ~] = size(bus);
 92
 93shouldIPlotEverything = false;
 94whatBusShouldIPlot = min(size(bus, 1), 11);
 95
 96if nargin < 2 %no participation factors so keep the current load profile
 97    participation = bus(:,PD)./sum(bus(:,PD));
 98else
 99    
100    participation = participation(:);%(:) forces column vector
101
102    if length(participation) ~= numBuses, %improper number of participations given
103        if length(participation) == 1 && participation > 0 && isinteger(participation),%assume bus number is specified instead
104            participation = (1:numBuses)'==participation;
105        else
106            if verbose, fprintf('\t[Info]\tParticipation Factors improperly specified.\n\t\t\tKeeping Current Loading Profile.\n'); end
107            participation = bus(:,PD)./sum(bus(:,PD));
108        end
109    end
110end
111
112[i2e, bus, gen, branch] = ext2int(bus, gen, branch);
113e2i = sparse(max(i2e), 1);
114e2i(i2e) = (1:size(bus, 1))';
115%loadvarloc_i = e2i(loadvarloc);
116participation_i = participation(e2i(i2e));
117
118participation_i = participation_i ./ sum(participation_i); %normalize
119
120
121    
122    
123%% get bus index lists of each type of bus
124[ref, pv, pq] = bustypes(bus, gen);
125
126
127
128if any(isnan(participation_i)), %could happen if no busses had loads
129    participation_i = zeros(length(participation_i), 1);
130    participation_i(pq) = 1/numel(participation_i(pq));
131end
132
133%% generator info
134on = find(gen(:, GEN_STATUS) > 0);      %% which generators are on?
135gbus = gen(on, GEN_BUS);                %% what buses are they at?
136
137%% form Ybus matrix
138[Ybus, ~, ~] = makeYbus(baseMVA, bus, branch);
139
140
141%% initialize parameters
142% set lambda to be increasing
143flag_lambdaIncrease = true;  % flag indicating lambda is increasing or decreasing
144
145%get all QP ratios
146initQPratio = bus(:,QD)./bus(:,PD);
147if any(isnan(initQPratio)), 
148    if verbose > 1, fprintf('\t[Warning]:\tLoad real power at bus %d is 0. Q/P ratio will be fixed at 0.\n', find(isnan(initQPratio)));  end
149    initQPratio(isnan(initQPratio)) = 0;
150end
151
152
153lambda0 = 2; 
154lambda = lambda0;
155Vm = ones(size(bus, 1), 1);          %% flat start
156Va = bus(ref(1), VA) * Vm;
157V  = Vm .* exp(1i* pi/180 * Va);
158V(gbus) = gen(on, VG) ./ abs(V(gbus)).* V(gbus);
159
160pointCnt = 0;
161
162%% do voltage correction (ie, power flow) to get initial voltage profile
163lambda_predicted = lambda;
164V_predicted = V;
165[V, lambda, success, ~] = cpf_correctVoltage(baseMVA, bus, gen, Ybus, V_predicted, lambda_predicted, initQPratio, participation_i);
166%% record data
167pointCnt = pointCnt + 1;
168
169
170%%------------------------------------------------
171% do cpf prediction-correction iterations
172%%------------------------------------------------
173t0 = clock;
174%% --- Start Phase 1: voltage prediction-correction (lambda increasing)
175if verbose > 0
176    fprintf('Start Phase 1: voltage prediction-correction (lambda increasing).\n');
177end
178i = 0;
179
180predicted_list = [];
181corrected_list = [];
182
183Vpr = [];
184lambda_pr = [];
185V_corr = [];
186lambda_corr = [];
187correctionIters = []; 
188slopes = [];
189
190stepSize = sigmaForLambda;
191minStepSize = 0.01;
192maxStepSize = 10;
193finished = false;
194
195while i < max_iter
196    %% update iteration counter
197    i = i + 1;
198    
199    % save good data
200    V_saved = V;
201    lambda_saved = lambda;
202    
203    if i<4, 
204        %% do voltage prediction to find predicted point (predicting voltage)
205        [V_predicted, lambda_predicted, J] = cpf_predict(Ybus, ref, pv, pq, V, lambda, stepSize, 1, initQPratio, participation_i, flag_lambdaIncrease);
206    else
207        [V_predicted, lambda_predicted] = cpf_predict_voltage(V_corr, lambda_corr, lambda, stepSize, ref, pv, pq);
208    end
209    %% do voltage correction to find corrected point
210    [V, lambda, success, iterNum] = cpf_correctVoltage(baseMVA, bus, gen, Ybus, V_predicted, lambda_predicted, initQPratio, participation_i);
211    
212
213    %% calculate slope (dP/dLambda) at current point
214    [slope, continuationBus] = max(abs(V-V_saved)  ./ (lambda-lambda_saved)); %calculate maximum slope at current point.
215    
216    
217    if success == false  && stepSize > minStepSize,
218        
219        stepSize = stepSize * 0.3;
220        
221        if verbose, fprintf('\t\tdidnt solve; changed stepsize to: %f\n', stepSize); end
222%       success = true;
223        i = i-1;
224        V = V_saved;
225        lambda = lambda_saved;
226        continue;
227    end
228%   
229    error = abs(V-V_predicted)./abs(V);
230%   try
231        if slope< 10^-10 || nnz(error) == 0, finished = true; break; end
232%   catch
233%       keyboard
234%   end
235    if mean(error) == 0, break; end
236    
237    if abs(log(mean(error)/0.0001)) > 1 && mean(error)>0,
238        stepSize = max(stepSize - 0.1*log( mean(error)/0.0001), minStepSize);
239        stepSize = min(stepSize, maxStepSize);
240        if verbose, fprintf('\t\tchanged stepsize to: %f\n', stepSize); end
241    end
242    
243    
244    %% instead of using condition number as criteria for switching between
245    %% modes...
246    %%    if rcond(J) <= condNumThresh_Phase1 | success == false % Jacobian matrix is ill-conditioned, or correction step fails
247    %% ...we use PV curve slopes as the criteria for switching modes
248    if abs(slope) >= slopeThresh_Phase1 || success == false % Approaching nose area of PV curve, or correction step fails
249        % restore good data
250        V = V_saved;
251        lambda = lambda_saved;
252        
253        if verbose > 0
254            if ~success, 
255                if ~isempty(strfind(lastwarn, 'singular')), 
256                    fprintf('\t[Info]:\tMatrix is singular. Aborting Correction.\n'); 
257                    lastwarn('No error');
258                    break;
259                else
260                    fprintf('\t[Info]:\tLambda correction fails.\n'); 
261                end
262            else
263                fprintf('\t[Info]:\tApproaching nose area of PV curve.\n');
264            end
265        end
266        break;
267    else
268        if verbose > 2
269            fprintf('\nVm_predicted\tVm_corrected\n');
270            [[abs(V_predicted);lambda_predicted] [abs(V);lambda]]
271        end
272
273        %% record data
274        pointCnt = pointCnt + 1;
275        Vpr = [Vpr, V_predicted];
276        lambda_pr = [lambda_pr, lambda_predicted];
277        V_corr = [V_corr, V];
278        lambda_corr = [lambda_corr, lambda];
279        correctionIters = [correctionIters, iterNum];
280        
281        predicted_list(:, pointCnt-1) = [V_predicted;lambda_predicted];
282        corrected_list(:, pointCnt) = [V;lambda];
283        
284        if shouldIPlotEverything,
285%           hold on; scatter(lambda, slope); hold off;
286            plot(lambda_corr, abs(V_corr(whatBusShouldIPlot,:)), 'o-');
287            hold on; scatter(lambda_corr, abs(Vpr(whatBusShouldIPlot,:)), 'r'); hold off
288            pause(0.01);
289        end
290    end
291end
292
293
294% fprintf('Average prediction error for voltage: %f\n', mean(mean( abs( V_corr - Vpr)./abs(V_corr))));
295% fprintf('Avg num of iterations: %f\n', mean(correctionIters));
296pointCnt_Phase1 = pointCnt; % collect number of points obtained at this phase
297if verbose > 0
298    fprintf('\t[Info]:\t%d data points contained in phase 1.\n', pointCnt_Phase1);
299end
300
301
302
303
304
305
306
307
308%% --- Switch to Phase 2: lambda prediction-correction (voltage decreasing)
309if verbose > 0
310    fprintf('Switch to Phase 2: lambda prediction-correction (voltage decreasing).\n');
311end
312k = 0;
313
314correctionIters2 = [];
315predictionErrors2 = [];
316
317startSlope = slope;
318stepSize = sigmaForVoltage;
319while k < max_iter && ~finished
320    %% update iteration counter
321    k = k + 1;
322
323    % save good data
324    V_saved = V;
325    lambda_saved = lambda;
326
327    %% do lambda prediction to find predicted point (predicting lambda)
328    
329    [V_predicted, lambda_predicted, J] = cpf_predict(Ybus, ref, pv, pq, V, lambda, stepSize, [2, continuationBus], initQPratio, participation_i,flag_lambdaIncrease);
330
331    %% do lambda correction to find corrected point
332    Vm_assigned = abs(V_predicted);
333    
334    [V, lambda, success, iterNum] = cpf_correctLambda(baseMVA, bus, gen, Ybus, Vm_assigned, V_predicted, lambda_predicted, initQPratio, participation_i, ref, pv, pq, continuationBus);
335    
336%   
337%   if success == false  && stepSize > minStepSize, 
338%       stepSize = stepSize * 0.3;  
339%       if verbose, fprintf('\t\tdidnt solve; changed stepsize to: %f\n', stepSize); end
340%       i = i-1;
341%       V = V_saved;
342%       lambda = lambda_saved;
343%       continue;
344%   end
345    
346    error = mean(abs(V-V_predicted)./abs(V));
347    
348
349    %% calculate slope (dP/dLambda) at current point
350    [slope, continuationBus] = max(abs(V-V_saved)  ./ (lambda-lambda_saved)); %calculate maximum slope at current point.
351    slopes = [slopes slope];
352%   
353%   proposedStepSize = min( max(stepSize - 0.03*log( mean(error)/0.0001), minStepSize), maxStepSize);
354%   
355%   stepSize = startSlope/slope^3 * proposedStepSize + (1-startSlope/slope^3) * minStepSize;
356%   
357    if slope< 10^-10 || error == 0, 
358        finished = true; break; end
359    
360    %% instead of using condition number as criteria for switching between
361    %% modes...
362    %%    if rcond(J) >= condNumThresh_Phase2 | success == false % Jacobian matrix is good-conditioned, or correction step fails
363    %% ...we use PV curve slopes as the criteria for switching modes
364    if abs(slope) <= slopeThresh_Phase2 || success == false % Leaving nose area of PV curve, or correction step fails
365        % restore good data
366        V = V_saved;
367        lambda = lambda_saved;
368
369        %% ---change to voltage prediction-correction (lambda decreasing)
370        if verbose > 0
371            if ~success, 
372                if ~isempty(strfind(lastwarn, 'singular'))
373                    fprintf('\t[Info]:\tMatrix is singular. Aborting Correction.\n');
374                    lastwarn('No error');
375                    break;
376                else
377                    fprintf('\t[Info]:\tLambda correction fails.\n');
378                end
379            else
380                fprintf('\t[Info]:\tLeaving nose area of PV curve.\n');
381            end
382        end
383        break;
384    else
385        if verbose > 2
386            fprintf('\nVm_predicted\tVm_corrected\n');
387            [[abs(V_predicted);lambda_predicted] [abs(V);lambda]]
388        end
389
390        %% record data
391        pointCnt = pointCnt + 1;
392        
393        Vpr = [Vpr, V_predicted];
394        lambda_pr = [lambda_pr, lambda_predicted];
395        V_corr = [V_corr, V];
396        lambda_corr = [lambda_corr, lambda];
397        
398        correctionIters2 = [correctionIters2 iterNum];
399        predictionErrors2 = [predictionErrors2 error];
400        
401        predicted_list(:, pointCnt-1) = [V_predicted;lambda_predicted];
402        corrected_list(:, pointCnt) = [V;lambda];
403        
404        if shouldIPlotEverything,
405%           hold on; scatter(lambda, slope); hold off;
406            plot(lambda_corr, abs(V_corr(whatBusShouldIPlot,:)), 'o-');
407            hold on; scatter(lambda_corr, abs(Vpr(whatBusShouldIPlot,:)), 'r'); hold off;
408        end
409    end
410end
411
412
413
414
415% fprintf('Average prediction error for voltage: %f\n', mean(predictionErrors2));
416% fprintf('Avg num of iterations: %f\n', mean(correctionIters2));
417
418pointCnt_Phase2 = pointCnt - pointCnt_Phase1; % collect number of points obtained at this phase
419if verbose > 0
420    fprintf('\t[Info]:\t%d data points contained in phase 2.\n', pointCnt_Phase2);
421end
422
423
424
425
426
427
428
429
430
431%% --- Switch to Phase 3: voltage prediction-correction (lambda decreasing)
432if verbose > 0
433    fprintf('Switch to Phase 3: voltage prediction-correction (lambda decreasing).\n');
434end
435% set lambda to be decreasing
436flag_lambdaIncrease = false; 
437i = 0;
438
439
440
441while i < max_iter && ~finished
442    break;
443    %% update iteration counter
444    i = i + 1;
445    
446    %% do voltage prediction to find predicted point (predicting voltage)
447    [V_predicted, lambda_predicted, J] = cpf_predict(Ybus, ref, pv, pq, V, lambda, stepSize, 1, initQPratio, participation_i, flag_lambdaIncrease);
448    
449    %% do voltage correction to find corrected point
450    [V, lambda, success, iterNum] = cpf_correctVoltage(baseMVA, bus, gen, Ybus, V_predicted, lambda_predicted, initQPratio, participation_i);
451
452    %% calculate slope (dP/dLambda) at current point
453    slope = min( abs( V-V_saved)./(lambda-lambda_saved));
454    
455    
456    if slope< 10^-10 || error == 0, finished = true; break; end
457%     slope = abs(V(loadvarloc_i) - V_saved(loadvarloc_i))/(lambda - lambda_saved);
458
459    if lambda < 0 % lambda is less than 0, then stops CPF simulation
460        if verbose > 0
461            fprintf('\t[Info]:\tlambda is less than 0.\n\t\t\tCPF finished.\n');
462        end
463        break;
464    end
465    
466    %% instead of using condition number as criteria for switching between
467    %% modes...
468    %%    if rcond(J) <= condNumThresh_Phase3 | success == false % Jacobian matrix is ill-conditioned, or correction step fails
469    %% ...we use PV curve slopes as the criteria for switching modes
470    if success == false % voltage correction step fails.
471        
472        if verbose > 0
473            if ~isempty(strfind(lastwarn, 'singular'))
474                fprintf('\t[Info]:\tMatrix is singular. Aborting Correction.\n');
475                lastwarn('No error');
476                break;
477            else
478                fprintf('\t[Info]:\tVoltage correction step fails..\n');
479            end
480        end
481        break;
482    else
483        if verbose > 2
484            fprintf('\nVm_predicted\tVm_corrected\n');
485            [[abs(V_predicted);lambda_predicted] [abs(V);lambda]]
486        end
487
488        %% record data
489        pointCnt = pointCnt + 1;
490        
491        Vpr = [Vpr, V_predicted];
492        lambda_pr = [lambda_pr, lambda_predicted];
493        V_corr = [V_corr, V];
494        lambda_corr = [lambda_corr, lambda];
495        
496        predicted_list(:, pointCnt-1) = [V_predicted;lambda_predicted];
497        corrected_list(:, pointCnt) = [V;lambda];
498        
499        if shouldIPlotEverything,
500            plot(lambda_corr, abs(V_corr(whatBusShouldIPlot,:)), 'o-');
501            hold on; scatter(lambda_corr, abs(Vpr(whatBusShouldIPlot,:)), 'r'); hold off
502        end
503    end
504end
505
506
507pointCnt_Phase3 = pointCnt - pointCnt_Phase2 - pointCnt_Phase1; % collect number of points obtained at this phase
508if verbose > 0
509    fprintf('\t[Info]:\t%d data points contained in phase 3.\n', pointCnt_Phase3);
510end
511
512et = etime(clock, t0);
513
514if i == max_iter,
515    fprintf('\t[Info] Max iterations hit.\n');
516end
517
518if  ~exist('corrected_list','var') | ~exist('predicted_list','var'),
519
520    max_lambda = NaN;
521    predicted_list = [];
522    corrected_list = [];
523    combined_list = [];
524    success==false;
525end
526
527if ~exist('max_lambda', 'var'),
528    max_lambda = NaN;
529    success==false;
530end
531
532if ~isempty(predicted_list) && ~isempty(corrected_list),
533    %% combine the prediction and correction data in the sequence of appearance
534    % NOTE: number of prediction data is one less than that of correction data
535    predictedCnt = size(predicted_list, 2);
536    combined_list(:, 1) = corrected_list(:, 1);
537    for i = 1:predictedCnt
538        combined_list(:, 2*i)     = predicted_list(:, i);
539        combined_list(:, 2*i+1)   = corrected_list(:, i+1);
540    end
541
542    %% convert back to original bus numbering & print results
543    [bus, gen, branch] = int2ext(i2e, bus, gen, branch);
544
545    %% add bus number as the first column to the prediction, correction, and combined data list
546    nb          = size(bus, 1);
547    max_lambda  = max(corrected_list(nb+1, :));
548    predicted_list = [[bus(:, BUS_I);0] predicted_list];
549    corrected_list = [[bus(:, BUS_I);0] corrected_list];
550    combined_list  = [[bus(:, BUS_I);0] combined_list];
551else
552    combined_list = [];
553end
554
555if verbose > 1
556    Vm_corrected = abs(corrected_list);
557    Vm_predicted = abs(predicted_list);
558    Vm_combined  = abs(combined_list);
559    Vm_corrected
560    Vm_predicted
561    Vm_combined
562    pointCnt_Phase1
563    pointCnt_Phase2
564    pointCnt_Phase3
565    pointCnt
566end
567
568if nargout == 1,
569    results.max_lambda = max_lambda;
570    results.V_pr = Vpr;
571    results.lambda_pr = lambda_pr;
572    results.V_corr = V_corr;
573    results.lambda_corr = lambda_corr;
574    results.success = success;
575    results.et = et;
576    
577    max_lambda = results; %return a single struct
578end

580%% Changelog - Anton Lodder - 2013.3.27
581% I implemented participation factor loading to allow all buses to
582% participate in load increase as a function of lambda.
583%
584% * Participation factors should be given as a vector, one value for each
585%   bus.
586% * if only one value is given, it is assumed that the value is a bus
587%   number rather than a  participation factor, and all other buses get a
588%   participation factor of zero (point two)
589% * any buses with zero participation factor will remain at their initial
590%   load level
591% * from the previous two bullets: backwards compatibility is maintained
592%   while allowing increased functionality
593% * if participation is not a valid bus number (eg float, negative number),
594%   maintains given bus loading profile.
595% * if no participation factors are given, maintain given bus loading
596%   profile.
